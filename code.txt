import serial
import paho.mqtt.client as mqtt
import time
# Configuration du port série (remplacez '/dev/ttyUSB0' par le port de votre Arduino)
SERIAL_PORT = "/dev/ttyUSB0"
BAUD_RATE = 9600
# Configuration MQTT
BROKER = "mqtt.eclipseprojects.io" # Remplacez par votre broker
PORT = 1883
TOPIC = "moniot/luminosite"
CLIENT_ID = "Raspberry_LDR"
# Connexion série et MQTT
try:
ser = serial.Serial(SERIAL_PORT, BAUD_RATE, timeout=1)
client = mqtt.Client(CLIENT_ID)
client.connect(BROKER, PORT, 60)
except Exception as e:
print(f"Erreur de connexion : {e}")
exit()

3

try:
while True:
if ser.in_waiting > 0:
# Lecture des données de l'Arduino
data = ser.readline().decode('utf-8').strip()
print(f"Luminosité : {data}")
# Envoi des données au broker MQTT
client.publish(TOPIC, data)
time.sleep(1)
except KeyboardInterrupt:
print("Arrêt du programme.")
finally:
ser.close()
client.disconnect()
